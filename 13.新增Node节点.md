初始化

``` bash
source /opt/k8s/bin/environment.sh
for node_ip in ${WORKER_NODES[@]}
  do
    echo ">>> ${node_ip}"
    ssh root@${node_ip} "apt-get install -y conntrack ipvsadm ntp ntpdate ipset jq iptables curl sysstat libseccomp2 && modprobe ip_vs "
  done
```



将kube-proxy,kubelet拷贝到所有纯worker节点：

```bash
    cd /opt/k8s/work
    source /opt/k8s/bin/environment.sh
    for node_ip in ${WORKER_NODES[@]}
      do
        echo ">>> ${node_ip}"
        scp kubernetes/server/bin/{kube-proxy,kubelet} root@${node_ip}:/opt/k8s/bin/
        ssh root@${node_ip} "chmod +x /opt/k8s/bin/*"
      done
```
